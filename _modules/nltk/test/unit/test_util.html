import pytest

from nltk.util import everygrams, usage



[docs]def test_usage_with_self(capsys):
    class MyClass:
        def kwargs(self, a=1):
            ...

        def no_args(self):
            ...

        def pos_args(self, a, b):
            ...

        def pos_args_and_kwargs(self, a, b, c=1):
            ...

    usage(MyClass)

    captured = capsys.readouterr()
    assert captured.out == (
        "MyClass supports the following operations:\n"
        "  - self.kwargs(a=1)\n"
        "  - self.no_args()\n"
        "  - self.pos_args(a, b)\n"
        "  - self.pos_args_and_kwargs(a, b, c=1)\n"
    )
